!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
! paramters - from CERN note 
!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!

aw = 1.5/1000.0
bw = 15.0/1000.0
!dx = 1.5/1000.0
!x0 = 10.0/1000.0
!y0 = 5.0/1000.0

dx = 0.5/1000.0
x0 = 15/1000.0
y0 = 1.0/1000.0



rfe = 60.0/1000.0
dfe = 40.0/1000.0

rbc = (rfe+dfe)*2.0
rbc = 1.0   !match with cern

nc = 16   ! 9 turns

!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!




*do,i,1,nc
	!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
	! make the rectangle of the ith cable
	!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!!
	*set,xc,x0+dx*(i-1)+aw*(i-1)+aw/2.0
	*set,yc,y0+bw/2.0
	*set,h,bw
	*set,w,aw

	!resets numbering to highest number currently
	alls
	numcmp,area
	numcmp,line
	numcmp,kp
	*get,knum,kp,,num,max
	*get,lnum,line,,num,max
	*get,anum,area,,num,max
	numstr,kp,knum
	numstr,line,lnum
	numstr,area,anum


	!make keypoints
	k,knum+1,xc-w/2.0,yc-h/2.0,
	k,knum+2,xc-w/2.0,yc+h/2.0,
	k,knum+3,xc+w/2.0,yc+h/2.0,
	k,knum+4,xc+w/2.0,yc-h/2.0,

	!make lines
	l,knum+1,knum+2
	l,knum+2,knum+3
	l,knum+3,knum+4
	l,knum+4,knum+1

	!make retangular area
	lsel,s,line,,lnum+1,lnum+4
	al,all
	asel,s,area,,anum+1
	cm,cond%i%,area       !name the area
*enddo


!name conductor 
cmsel,s,cond1
*do,i,2,nc
	cmsel,a,cond%i%
*enddo
cm,cond,area
